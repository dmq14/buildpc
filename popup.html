var a = "1.500.500đ"; var b = "2.500.000đ"; // Loại bỏ ký tự không phải là số và chuyển đổi thành số var numberA = parseFloat(a.replace(/[^\d]/g, '')); var numberB = parseFloat(b.replace(/[^\d]/g, '')); // Thực hiện phép cộng var result = numberA + numberB; var formattedResult = result.toLocaleString( { style: 'currency', currency: 'VND' }); formattedResult = formattedResult.replace('$', '') + 'đ'; console.log(formattedResult); // Output: "4,000,500.00 ₫" console.log(result); // Kết quả sẽ là 4000500
